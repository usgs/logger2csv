16  /**

25  public class Logger2csv {
26  
27      private static final int M_TO_S = 60;
28      private static final int DAY_TO_S = 24 * 60 * M_TO_S;
29      private static final int M_TO_MS = 60 * 1000;
30  
31      public static final String DEFAULT_CONFIG_FILENAME = "logger2csv.config";
32      public static final int DEFAULT_INTERVAL_M = M_TO_S;
33  
34      private static final String EXAMPLE_CONFIG_FILENAME = "logger2csv-example.config";
35      private static final Logger LOGGER = LoggerFactory.getLogger(Logger2csv.class);
36  
37      private ConfigFile configFile;
38      private List<DataLogger> loggers;
39      private int interval;
40  
41      public Logger2csv(ConfigFile configFile) {
42          LOGGER.info("Launching Logger2csv ({})", Logger2csvVersion.VERSION_STRING);
43  
44          this.configFile = configFile;
45          this.interval = configFile.getInt("interval", DEFAULT_INTERVAL_M);
46          loggers = getLoggers();
47      }
48  
49      private List<DataLogger> getLoggers() {
50          List<DataLogger> loggers = new ArrayList<DataLogger>();
51  
52          for (String station : configFile.getList("station")) {
53              ConfigFile config = configFile.getSubConfig(station, true);
54              config.put("name", station);
55  
56              try {
57                  loggers.add(new DataLogger(config));
58              } catch (UnknownHostException e) {
59                  LOGGER.error("Cannot find host \"{}\". I'll skip it this time.", config.getString("address"));
60              } catch (IOException e) {
61                  LOGGER.error(e.getMessage());
62              }
63          }
64          return loggers;
65      }
66  
67      public void poll(DataLogger logger, String table) {
68          FileDataReader fileReader = new FileDataReader(logger, table);
69          WebDataReader webReader = new WebDataReader(logger, table);
70          FileDataWriter fileWriter = new FileDataWriter(logger, table);
71          LOGGER.debug("Polling {}.{}", logger.name, table);
72  
73          int lastRecord;
74          try {
75              lastRecord = fileReader.findLastRecord();
76          } catch (IOException e1) {
77              LOGGER.error("Cannot parse file on disk for {}.{}, I'll skip it this time. File corrupt?", logger.name,
78                      table);
79              return;
80          }
81  
82          Iterator<String[]> results;
83          try {
84              if (lastRecord > 0)
85                  results = webReader.since_record(lastRecord);
86              else
87                  results = webReader.backFill(logger.backfill * DAY_TO_S);
88          } catch (IOException e) {
89              LOGGER.error("Cannot read new records from {}.{}, I'll skip it this time.", logger.name, table);
90              return;
91          }
92  
93          try {
94              fileWriter.write(results, lastRecord);
95          } catch (ParseException e) {
96              LOGGER.error("Cannot parse logger response. Skipping {}", logger.name);
97              return;
98          } catch (IOException e) {
99              LOGGER.error("Cannot write to datafile for {}.{}.", logger.name, table);
100             return;
101         }
102 
103     }
104 
105     public void pollAllOnce() {
106         for (DataLogger l : loggers) {
107             Iterator<String> it = l.getTableIterator();
108             while (it.hasNext()) {
109                 String table = it.next();
110                 poll(l, table);
111             }
112         }
113     }
114 
115     private void pollAllCountinous() {
116         while (true) {
117             pollAllOnce();
118             try {
119                 Thread.sleep(interval * M_TO_MS);
120             } catch (InterruptedException ignore) {
121             }
122         }
123     }
124 
125     public static void main(String[] args) {
126         Logger2csvArgs config = new Logger2csvArgs(args);
127 
128         ConfigFile cf = null;
129         try {
130             cf = new ConfigFile(config.configFileName);
131         } catch (FileNotFoundException e) {
132             LOGGER.warn("Can't parse config file " + config.configFileName + ". Try using the --help flag.");
133             System.exit(1);
134         }
135 
136         Logger2csv logger2csv = new Logger2csv(cf);
137         if (config.persistent)
138             logger2csv.pollAllCountinous();
139         else
140             logger2csv.pollAllOnce();
141     }

7   package gov.usgs.volcanoes.logger2csv;
8   
9   import gov.usgs.volcanoes.core.configfile.ConfigFile;
10  import gov.usgs.volcanoes.core.configfile.parser.CsvFormatParser;
11  
12  import org.apache.commons.csv.CSVFormat;
13  import org.slf4j.Logger;
14  import org.slf4j.LoggerFactory;
15  
16  import java.io.IOException;
17  import java.text.ParseException;
18  import java.text.SimpleDateFormat;
19  import java.util.TimeZone;
20  
21  /**
22   * A class to hold configuration and utility methods for a single data logger
23   * 
24   * @author Tom Parker
25   * 
26   *         I waive copyright and related rights in the this work worldwide
27   *         through the CC0 1.0 Universal public domain dedication.
28   *         https://creativecommons.org/publicdomain/zero/1.0/legalcode
29   */
30  public abstract class DataLogger {
31    private static final Logger LOGGER = LoggerFactory.getLogger(Logger2csv.class);
32  
33    public static final String DEFAULT_FILE_PATH_FORMAT = "yyyy/MM";
34    public static final String DEFAULT_FILE_SUFFIX_FORMAT = "-yyyyMMdd'.csv'";
35    public static final String DEFAULT_PATH_ROOT = "data";
36    public static final int DEFAULT_BACKFILL = 2;
37    public static final boolean DEFAULT_QUOTE_FIELDS = false;
38    public static final String TOA5_DATE_FORMAT = "yyyy-MM-dd HH:mm:ss";
39    public static final CSVFormat DEFAULT_CSV_FORMAT = CSVFormat.RFC4180;
40    public static final String DEFAULT_COMMENT_CHAR = "#";
41  
42    public final String name;
43    public final int backfill;
44    public final String pathRoot;
45    public final String address;
46    public final boolean quoteFields;
47    public final int headerCount;
48    public final CSVFormat csvFormat;
49    public final String commentChar;
50  
51    protected final SimpleDateFormat filePathFormat;
52    protected final SimpleDateFormat fileSuffixFormat;
53    protected final SimpleDateFormat dateFormat;
54    protected final SimpleDateFormat csvDateFormat;
55  
56    public DataLogger(ConfigFile config, int headerCount) throws IOException, ParseException {
57      this.headerCount = headerCount;
58      TimeZone.setDefault(TimeZone.getTimeZone("UTC"));
59  
60      name = config.getString("name");
61      if (name == null)
62        throw new IOException("Station name must be specified in Config.");
63      LOGGER.debug("creating logger {}", name);
64  
65      address = config.getString("address");
66      if (address == null)
67        throw new IOException("Station address must be specified in Config.");
68  
69      backfill = config.getInt("backfill", DEFAULT_BACKFILL);
70      pathRoot = config.getString("pathRoot", DEFAULT_PATH_ROOT);
71      quoteFields = config.getBoolean("quoteFields", DEFAULT_QUOTE_FIELDS);
72  
73      String path = config.getString("filePathFormat", DEFAULT_FILE_PATH_FORMAT);
74      filePathFormat = new SimpleDateFormat(path);
75  
76      String suffix = config.getString("fileSuffixFormat", DEFAULT_FILE_SUFFIX_FORMAT);
77      fileSuffixFormat = new SimpleDateFormat(suffix);
78  
79      dateFormat = new SimpleDateFormat(TOA5_DATE_FORMAT);
80  
81      csvFormat = config.getObject("csvFormat", new CsvFormatParser());
82  
83      commentChar = config.getString("commentChar", DEFAULT_COMMENT_CHAR);
84  
85      String csvDateFormatString = config.getString("csvDateFormat", dateFormat.toPattern());
86      csvDateFormat = new SimpleDateFormat(csvDateFormatString);
87    }
88  }

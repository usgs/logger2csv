3   import java.io.File;
4   import java.io.IOException;
5   import java.text.ParseException;
6   import java.text.SimpleDateFormat;
7   import java.util.Date;
8   import java.util.Iterator;
9   import java.util.List;
10  import java.util.TimeZone;
11  
12  import org.slf4j.Logger;
13  import org.slf4j.LoggerFactory;
14  
15  import gov.usgs.volcanoes.util.configFile.ConfigFile;
16  
17  /**
18   * A class to hold configuration and utility methods for a single data logger
19   *

26  public class DataLogger {
27  	private static final Logger LOGGER = LoggerFactory.getLogger(Logger2csv.class);
28  
29  	public static final String DEFAULT_FILE_PATH_FORMAT = "yyyy/MM";
30  	public static final String DEFAULT_FILE_SUFFIX_FORMAT = "-yyyyMMdd";
31  	public static final String DEFAULT_PATH_ROOT = "data";
32  	public static final int DEFAULT_BACKFILL = 2;
33  	public static final boolean DEFAULT_QUOTE_FIELDS = false;
34  	public static final String TOA5_DATE_FORMAT = "yyyy-MM-dd HH:mm:ss";
35  	public static final int HEADER_COUNT = 4;
36  
37  	public final String name;
38  	public final int backfill;
39  	public final String pathRoot;
40  	public final String address;
41  	public final boolean quoteFields;
42  	
43  	private final SimpleDateFormat filePathFormat;
44  	private final SimpleDateFormat fileSuffixFormat;
45  	private final List<String> tables;
46  	private final SimpleDateFormat dateFormat;
47  
48  	public DataLogger(ConfigFile config) throws IOException {
49  		TimeZone.setDefault(TimeZone.getTimeZone("UTC"));
50  
51  		name = config.getString("name");
52  		if (name == null)
53  			throw new IOException("Station name must be specified in Config.");
54  		LOGGER.debug("creating logger {}", name);
55  
56  		address = config.getString("address");
57  		if (address == null)
58  			throw new IOException("Station address must be specified in Config.");
59  
60  		backfill = config.getInt("backfill", DEFAULT_BACKFILL);
61  		pathRoot = config.getString("pathRoot", DEFAULT_PATH_ROOT);
62  		quoteFields = config.getBoolean("quoteFields", DEFAULT_QUOTE_FIELDS);
63  		tables = config.getList("table");
64  		if (tables == null)
65  			throw new IOException("No tables found for " + name);
66  
67  		String path = config.getString("filePathFormat", DEFAULT_FILE_PATH_FORMAT);
68  		filePathFormat = new SimpleDateFormat(path);
69  
70  		String suffix = config.getString("fileSuffixFormat", DEFAULT_FILE_SUFFIX_FORMAT);
71  		fileSuffixFormat = new SimpleDateFormat(suffix);
72  
73  		dateFormat = new SimpleDateFormat(TOA5_DATE_FORMAT);
74  	}
75  
76  	public Iterator<String> getTableIterator() {
77  		return tables.iterator();
78  	}
79  
80  	public Date parseDate(String date) throws ParseException {
81  		return dateFormat.parse(date);
82  	}
83  
84  	public String getFileName(String table, long timeMs) {
85  		StringBuilder sb = new StringBuilder();
86  		sb.append(pathRoot);
87  		sb.append("/");
88  		sb.append(name);
89  		sb.append("/");
90  		sb.append(filePathFormat.format(timeMs));
91  		sb.append("/");
92  		sb.append(name);
93  		sb.append("-");
94  		sb.append(table);
95  		sb.append(fileSuffixFormat.format(timeMs));
96  
97  		String filename = sb.toString().replace('/', File.separatorChar);
98  		return filename;
99  	}
100 }

23  public class FileDataReader {
24  	private static final Logger LOGGER = LoggerFactory.getLogger(Logger2csv.class);
25  
26  	public static final String FILE_EXTENSION = ".csv";
27  
28  	private static final long DAY_TO_MS = 24 * 60 * 60 * 1000;
29  	
30  	private final DataLogger logger;
31  	private final String table;
32  
33  	public FileDataReader(DataLogger logger, String table) {
34  		this.logger = logger;
35  		this.table = table;
36  	}
37  
38  	public int findLastRecord() throws IOException {
39  		LOGGER.debug("Finding last record for {}", logger.name);
40  		CSVReader reader = null;
41  
42  		int lr = -1;
43  		long timeMs = System.currentTimeMillis();
44  		long ancientMs = timeMs - logger.backfill * DAY_TO_MS;
45  		while (lr == -1 && timeMs > ancientMs) {
46  			String fileName = logger.getFileName(table, timeMs) + FILE_EXTENSION;
47  			LOGGER.debug("looking for {}", fileName);
48  			try {
49  			    Reader fr = new InputStreamReader(new FileInputStream(fileName), "UTF-8");
50  				reader = new CSVReader(fr);
51  				lr = -1;
52  				String[] nextLine;
53  
54  				// skip header lines at the top
55  				for (int i = 0; i < DataLogger.HEADER_COUNT; i++)
56  					nextLine = reader.readNext();
57  
58  				while ((nextLine = reader.readNext()) != null) {
59  					lr = Integer.parseInt(nextLine[1]);
60  				}
61  			} catch (FileNotFoundException dontCare) {
62  				LOGGER.debug("This is okay: {}", dontCare.getLocalizedMessage());
63  			} finally {
64  				if (reader != null)
65  					reader.close();
66  			}
67  			timeMs -= DAY_TO_MS;
68  		}
69  
70  		LOGGER.debug("found record {}.", lr);
71  		return lr;
72  	}
73  }

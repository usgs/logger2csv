7   package gov.usgs.volcanoes.logger2csv;
8   
9   import gov.usgs.volcanoes.core.configfile.ConfigFile;
10  import gov.usgs.volcanoes.logger2csv.campbell.CampbellDataLogger;
11  import gov.usgs.volcanoes.logger2csv.campbell.CampbellPoller;
12  
13  import java.io.IOException;
14  import java.text.ParseException;
15  
16  /**
17   * Types of data loggers I can talk to.
18   * 
19   * @author Tom Parker
20   *
21   */
22  public final class PollerFactory {
23    
24    private static enum LoggerType {
25    /** Designed to work with CampbellScientific CR850 and CR1000 */
26  	CAMPBELL, 
27  	
28  	/** Designed to work with MetOne E-BAM */
29  	EBAM
30  	
31    }
32    
33    // non-instantiatable
34    private PollerFactory() {}
35    
36    public static Poller getPoller(ConfigFile config) throws IOException, ParseException {
37      Poller poller = null;
38      LoggerType type;
39      try {
40        type = LoggerType.valueOf(config.getString("type").toUpperCase());
41      } catch (IllegalArgumentException e) {
42        throw new ParseException("Cannot find logger type.", -1);
43      }
44      
45      switch (type) {
46        case CAMPBELL:
47          CampbellDataLogger logger = new CampbellDataLogger(config);
48          poller = new CampbellPoller(logger);
49          break;
50        default:
51          throw new UnsupportedOperationException("Unknown type: " + type);
52      }
53      
54      return poller;
55    }
56  }

6   package gov.usgs.volcanoes.logger2csv.logger;
7   
8   import org.apache.commons.csv.CSVRecord;
9   
10  import java.text.ParseException;
11  import java.text.SimpleDateFormat;
12  import java.util.ArrayList;
13  import java.util.List;
14  import java.util.ListIterator;
15  
16  /**
17   * Holder class for  single CSV record.
18   * 
19   * @author Tom Parker
20   *
21   */
22  public class LoggerRecord {
23  
24    /**  time in ms as returned by Date.getTime()  */
25    public final long date;
26    
27    /** My CSVRecord */
28    public final CSVRecord record;
29  
30    /**
31     * Constructor.
32     * 
33     * @param date Time in ms as returned by Date.getTime()
34     * @param record My CSVRecord
35     */
36    public LoggerRecord(long date, CSVRecord record) {
37      this.date = date;
38      this.record = record;
39    }
40  
41    /**
42     * Utility method to create a new LoggerRecord.
43     * 
44     * @param record My CSVRecord
45     * @param dateFormat Format passed to SimpleDateFormat to parse the date field
46     * @param dateIndex Index of date filed
47     * @return A newly constructed LoggerRecord
48     * @throws ParseException when date cannot be parsed
49     */
50    public static LoggerRecord fromCSVRecord(CSVRecord record, SimpleDateFormat dateFormat,
51        int dateIndex) throws ParseException {
52      String dateString = record.get(dateIndex);
53      long date = dateFormat.parse(dateString).getTime();
54  
55      return new LoggerRecord(date, record);
56    }
57  
58    /**
59     * Utility method to create a list of LoggerRecords.
60     * 
61     * @param recordList List of CSVRecords
62     * @param dateFormat Format passed to SimpleDateFormat to parse the date field
63     * @param dateIndex Index of date filed
64     * @return A newly constructed list of loggerRecords
65     * @throws ParseException when any of the CSVRecords contain an unparsable date field
66     */
67    public static List<LoggerRecord> fromCSVList(ListIterator<CSVRecord> recordList,
68        SimpleDateFormat dateFormat, int dateIndex) throws ParseException {
69      List<LoggerRecord> loggerRecords = new ArrayList<LoggerRecord>();
70  
71      while (recordList.hasNext()) {
72        CSVRecord record = recordList.next();
73        String dateString = record.get(dateIndex);
74        long date = dateFormat.parse(dateString).getTime();
75        loggerRecords.add(new LoggerRecord(date, record));
76      }
77  
78      return loggerRecords;
79    }
80  }

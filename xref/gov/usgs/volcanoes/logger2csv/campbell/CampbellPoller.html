44    public CampbellPoller(CampbellDataLogger logger) {
45      this.logger = logger;

48    public void updateFiles() {
49      LOGGER.debug("Polling {}", logger.name);
50      Iterator<String> tableIt = logger.getTableIterator();
51      while (tableIt.hasNext()) {
52        String table = tableIt.next();
53        try {
54          updateTable(table);
55        } catch (PollerException e) {
56          LOGGER.error("Unable to update {}.{}, I'll try again next time. ({})", logger.name, table,
57              e.getLocalizedMessage());
58        }
59      }
60    }
61  
62    private void updateTable(String table) throws PollerException {
63      int lastRecordNum = findLastRecordNum(table);
64      ListIterator<CSVRecord> results = retrieveNewData(table, lastRecordNum);
65  
66      if (!results.hasNext()) {
67        LOGGER.info("No response when polling {}.{}", logger.name, table);
68        return;
69      }
70  
71      List<CSVRecord> headers = extractHeaders(results);
72  
73      if (!results.hasNext()) {
74        LOGGER.debug("No new data in table {}.{}", logger.name, table);
75        return;
76      }
77  
78      CSVRecord record = results.next();
79      if (recordMatches(record, lastRecordNum)) {
80        // do nothing and let the old value fall on the floor
81      } else {
82        // different record. Quick, put it back!
83        results.previous();
84      }
85  
86      if (!results.hasNext()) {
87        LOGGER.debug("no data found");
88        return;
89      }
90  
91      List<LoggerRecord> records;
92      SimpleDateFormat dateFormat = new SimpleDateFormat(CampbellDataLogger.DATE_FORMAT_STRING);
93      try {
94        records = LoggerRecord.fromCSVList(results, dateFormat, CampbellDataLogger.DATE_COLUMN);
95      } catch (ParseException e1) {
96        throw new PollerException("Cannot parse results. (" + e1.getLocalizedMessage() + ")");
97      }
98  
99      // write new data
100     FileDataWriter fileWriter = new FileDataWriter(logger.csvFormat, logger.getFilePattern(table));
101     fileWriter.addHeaders(headers);
102     try {
103       fileWriter.write(records.iterator());
104 //    } catch (ParseException e) {
105 //      LOGGER.error("Cannot parse logger response. Skipping {}. ({})", logger.name, e);
106 //      return;
107     } catch (IOException e) {
108       LOGGER.error("Cannot write to datafile for {}.{}.", logger.name, table);
109       return;
110     }
111 
112   }

114   private ListIterator<CSVRecord> retrieveNewData(String table, int lastRecordNum)
115       throws PollerException {
116     LOGGER.debug("Polling {}.{}", logger.name, table);
117 
118     ListIterator<CSVRecord> results;
119     if (lastRecordNum > 0)
120       results = since_record(lastRecordNum, table);
121     else
122       results = backFill((int) (logger.backfill * DAY_TO_S), table);
123 
124     return results;
125   }
126 
127   private boolean recordMatches(CSVRecord record, int recordNum) {
128     if (logger.parseRecordNum(record) == recordNum)
129       return true;
130     else
131       return false;
132   }
133 
134   private int findLastRecordNum(String table) throws PollerException {
135     FileDataReader fileReader = new FileDataReader(logger);
136     CSVRecord lastRecord = fileReader.findLastRecord(logger.getFilePattern(table));
137     if (lastRecord == null) {
138       LOGGER.debug("No recent data file was found.");
139       return -1;
140     } else {
141       LOGGER.debug("Most recent record num is {}", lastRecord.get(1));
142       return Integer.parseInt(lastRecord.get(1));
143     }
144   }
145 
146   private ListIterator<CSVRecord> since_record(int record, String table) throws PollerException {
147     LOGGER.info("Downloading new records from {}.{}.", logger.name, table);
148     return getResults("since-record", record, table);
149   }
150 
151   private ListIterator<CSVRecord> backFill(int backfillS, String table) throws PollerException {
152     LOGGER.info("Downloading all recent records from {}.{}.", logger.name, table);
153     return getResults("backfill", backfillS, table);
154   }

156   private ListIterator<CSVRecord> getResults(final String mode, final int p1, String table)
157       throws PollerException {
158     StringBuilder sb = new StringBuilder();
159     sb.append("http://");
160     sb.append(logger.address);
161     sb.append(":" + logger.port);
162     sb.append("/?command=DataQuery&uri=dl:");
163     sb.append(table);
164     sb.append("&mode=");
165     sb.append(mode);
166     sb.append("&format=TOA5");
167     sb.append("&p1=");
168     sb.append(p1);
169 
170     List<CSVRecord> records = null;
171     try {
172       // String url = sb.toString();
173       URL url = new URL(sb.toString());
174 
175       LOGGER.debug("Polling from {}", url);
176       CSVParser parser = CSVParser.parse(url, StandardCharsets.UTF_8, logger.csvFormat);
177       records = parser.getRecords();
178     } catch (IOException e) {
179       throw new PollerException(e);
180     }
181 
182     return records.listIterator();
183   }
184 
185   private List<CSVRecord> extractHeaders(Iterator<CSVRecord> iterator) {
186     List<CSVRecord> headers = new ArrayList<CSVRecord>();
187     for (int i = 0; i < logger.headerCount; i++) {
188       if (iterator.hasNext()) {
189         headers.add(iterator.next());
190       }
191     }
192 
193     return headers;
194   }

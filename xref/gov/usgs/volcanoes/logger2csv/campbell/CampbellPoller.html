26  /**
27   * A class to poll a single data logger.
28   * 
29   * @author Tom Parker
30   */
31  public final class CampbellPoller implements Poller {
32    private static final Logger LOGGER = LoggerFactory.getLogger(CampbellPoller.class);
33  
34    private final CampbellDataLogger logger;
35  
36    /**
37     * A Poller class for collecting data from a CompbellScientific data logger.
38     * 
39     * @param logger
40     *          The DataLogger to poll
41     */
42    public CampbellPoller(CampbellDataLogger logger) {
43      this.logger = logger;
44    }
45  
46    public void updateFiles() {
47      LOGGER.debug("Polling {}", logger.name);
48      Iterator<String> tableIt = logger.getTableIterator();
49      while (tableIt.hasNext()) {
50        String table = tableIt.next();
51        try {
52          updateTable(table);
53        } catch (IOException e) {
54          LOGGER.error("Unable to update {}.{}, I'll try again next time. ({})", logger.name, table,
55              e.getLocalizedMessage());
56        }
57      }
58    }
59  
60    private void updateTable(String table) throws IOException {
61      LOGGER.debug("Polling {}.{}", logger.name, table);
62  
63      int lastRecordNum = findLastRecordNum(table);
64      LOGGER.debug("Last record: {}", lastRecordNum);
65  
66      ListIterator<CSVRecord> results;
67      try {
68        results = retrieveNewData(table, lastRecordNum);
69      } catch (IOException e) {
70        LOGGER.error("Cannot retrieve data from {}.{}, I'll try again next time. ({})", logger.name,
71            table, e.getLocalizedMessage());
72        return;
73      }
74  
75      if (!results.hasNext()) {
76        LOGGER.info("No new data found in table {}.{}", logger.name, table);
77        return;
78      }
79  
80      List<CSVRecord> headers = extractHeaders(results);
81  
82      if (!results.hasNext()) {
83        LOGGER.debug("no data found");
84        return;
85      }
86      
87      CSVRecord record = results.next();
88      if (recordMatches(record, lastRecordNum)) {
89        // do nothing and let the old value fall on the floor
90      } else {
91        // different record. Quick, put it back!
92        results.previous();
93      }
94  
95      // write new data
96      FileDataWriter fileWriter = new CampbellWriter(logger, table);
97      fileWriter.addHeaders(headers);
98      fileWriter.write(results);
99    }
100 
101   private ListIterator<CSVRecord> retrieveNewData(String table, int lastRecordNum)
102       throws IOException {
103     ListIterator<CSVRecord> results;
104     if (lastRecordNum > 0)
105       results = since_record(lastRecordNum, table);
106     else
107       results = backFill((int) (logger.backfill * DAY_TO_S), table);
108 
109     return results;
110   }
111 
112   private boolean recordMatches(CSVRecord record, int recordNum) {
113     if (logger.parseRecordNum(record) == recordNum)
114       return true;
115     else
116       return false;
117   }
118 
119   private int findLastRecordNum(String table) throws IOException {
120     FileDataReader fileReader = new FileDataReader(logger);
121     CSVRecord lastRecord = fileReader.findLastRecord(logger.getFilePattern(table));
122     if (lastRecord == null) {
123       LOGGER.debug("No recent data file was found.");
124       return -1;
125     } else {
126       LOGGER.debug("Most recent record num is {}", lastRecord.get(1));
127       return Integer.parseInt(lastRecord.get(1));
128     }
129   }
130 
131   private ListIterator<CSVRecord> since_record(int record, String table) throws IOException {
132     LOGGER.info("Downloading new records from {}.{}.", logger.name, table);
133     return getResults("since-record", record, table);
134   }
135 
136   private ListIterator<CSVRecord> backFill(int backfillS, String table) throws IOException {
137     LOGGER.info("Downloading all recent records from {}.{}.", logger.name, table);
138     return getResults("backfill", backfillS, table);
139   }
140 
141   private ListIterator<CSVRecord> getResults(final String mode, final int p1, String table)
142       throws IOException {
143     StringBuilder sb = new StringBuilder();
144     sb.append("http://");
145     sb.append(logger.address);
146     sb.append("/?command=DataQuery&uri=dl:");
147     sb.append(table);
148     sb.append("&mode=");
149     sb.append(mode);
150     sb.append("&format=TOA5");
151     sb.append("&p1=");
152     sb.append(p1);
153 
154     // String url = sb.toString();
155     URL url = new URL(sb.toString());
156     
157     LOGGER.debug("Polling from {}", url);
158     CSVParser parser = CSVParser.parse(url, StandardCharsets.UTF_8, logger.csvFormat);
159     ListIterator<CSVRecord> iterator = parser.getRecords().listIterator();
160 
161     return iterator;
162   }
163 
164   private List<CSVRecord> extractHeaders(Iterator<CSVRecord> iterator) {
165     List<CSVRecord> headers = new ArrayList<CSVRecord>();
166     for (int i = 0; i < logger.headerCount; i++) {
167       if (iterator.hasNext()) {
168         headers.add(iterator.next());
169       }
170     }
171 
172     return headers;
173   }

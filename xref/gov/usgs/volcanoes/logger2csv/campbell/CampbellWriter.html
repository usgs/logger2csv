7   package gov.usgs.volcanoes.logger2csv.campbell;
8   
9   import gov.usgs.volcanoes.logger2csv.FileDataWriter;
10  
11  import org.apache.commons.csv.CSVRecord;
12  
13  import java.io.File;
14  import java.text.ParseException;
15  import java.text.SimpleDateFormat;
16  import java.util.Date;
17  
18  /**
19   * A class to write data from a single data logger.
20   * 
21   * @author Tom Parker
22   */
23  public final class CampbellWriter extends FileDataWriter {
24  
25    private final SimpleDateFormat loggerDateFormat;
26    private final SimpleDateFormat fileDateFormat;
27    private final CampbellDataLogger logger;
28  
29    /**
30     * Constructor.
31     * 
32     * @param filePattern Patter used to create filenames
33     */
34    public CampbellWriter(CampbellDataLogger logger, String table) {
35      super(logger.csvFormat);
36      this.logger = logger;
37      loggerDateFormat = new SimpleDateFormat(CampbellDataLogger.DATE_FORMAT_STRING);
38      fileDateFormat = new SimpleDateFormat(logger.getFilePattern(table));
39    }
40  
41    @Override
42    protected Date getDate(CSVRecord record) throws ParseException {
43      return loggerDateFormat.parse(record.get(CampbellDataLogger.DATE_COLUMN));
44    }
45  
46    @Override
47    protected File getFile(CSVRecord record) throws ParseException {
48      Date date = logger.parseDate(record);
49      return new File(fileDateFormat.format(date));
50    }
51  
52  }

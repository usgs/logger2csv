6   package gov.usgs.volcanoes.logger2csv.ebam;
7   
8   import gov.usgs.volcanoes.logger2csv.FileDataReader;
9   import gov.usgs.volcanoes.logger2csv.poller.Poller;
10  import gov.usgs.volcanoes.logger2csv.poller.PollerException;
11  
12  import org.apache.commons.csv.CSVRecord;
13  import org.slf4j.Logger;
14  import org.slf4j.LoggerFactory;
15  
16  import io.netty.bootstrap.Bootstrap;
17  import io.netty.channel.Channel;
18  import io.netty.channel.ChannelHandler;
19  import io.netty.channel.EventLoopGroup;
20  import io.netty.channel.nio.NioEventLoopGroup;
21  import io.netty.channel.socket.nio.NioSocketChannel;
22  
23  /**
24   * A class to poll a single eBAM data logger over the network.
25   * 
26   * @author Tom Parker
27   */
28  public final class EbamPoller implements Poller {
29    private static final Logger LOGGER = LoggerFactory.getLogger(EbamPoller.class);
30  
31    private final EbamDataLogger logger;
32  
33    /**
34     * A Poller class for collecting data from a CompbellScientific data logger.
35     * 
36     * @param logger The DataLogger to poll
37     */
38    public EbamPoller(EbamDataLogger logger) {
39      this.logger = logger;
40    }
41  
42    /**
43     * Update all files
44     */
45    public void updateFiles() {
46      LOGGER.debug("Polling {}", logger.name);
47      try {
48        updateFile(DataFile.DATA_LOG);
49        updateFile(DataFile.ERROR_LOG);
50      } catch (PollerException e) {
51        LOGGER.error("Unable to retrieve data from {}. ({})", logger.name, e);
52      }
53    }
54  
55    private int findLastRecordNum(DataFile dataFile) throws PollerException {
56      FileDataReader fileReader = new FileDataReader(logger);
57      CSVRecord lastRecord = fileReader.findLastRecord(logger.getFilePattern(dataFile));
58      if (lastRecord == null) {
59        LOGGER.debug("No recent data file was found.");
60        return -1;
61      } else {
62        int recordNumIdx = lastRecord.size() - 1;
63        LOGGER.debug("Most recent record num is {}", lastRecord.get(recordNumIdx));
64        return Integer.parseInt(lastRecord.get(recordNumIdx));
65      }
66    }
67  
68    private void updateFile(DataFile dataFile) throws PollerException {
69      EventLoopGroup group = new NioEventLoopGroup();
70      try {
71        Bootstrap b = new Bootstrap();
72        int recordIndex = findLastRecordNum(dataFile);
73        ChannelHandler handler = new EbamClientInitializer(logger, dataFile, recordIndex);
74        b.group(group).channel(NioSocketChannel.class).handler(handler);
75  
76        Channel ch;
77        try {
78          ch = b.connect(logger.address, logger.port).sync().channel();
79        } catch (InterruptedException e) {
80          throw new PollerException(e);
81        }
82  
83        LOGGER.debug("connected to {}.", logger.name);
84        try {
85          ch.closeFuture().sync();
86        } catch (InterruptedException e) {
87          e.printStackTrace();
88        }
89  
90      } finally {
91        group.shutdownGracefully();
92      }
93  
94    }
95  }

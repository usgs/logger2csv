6   package gov.usgs.volcanoes.logger2csv.ebam;
7   
8   import java.io.IOException;
9   
10  import io.netty.channel.ChannelInitializer;
11  import io.netty.channel.ChannelPipeline;
12  import io.netty.channel.socket.SocketChannel;
13  import io.netty.handler.codec.LineBasedFrameDecoder;
14  import io.netty.handler.codec.string.StringDecoder;
15  import io.netty.handler.codec.string.StringEncoder;
16  import io.netty.handler.timeout.ReadTimeoutHandler;
17  import io.netty.util.CharsetUtil;
18  
19  public class EbamClientInitializer extends ChannelInitializer<SocketChannel> {
20  
21  
22    private final EbamDataLogger logger;
23    private final DataFile dataFile;
24    private final int recordIndex;
25  
26    public EbamClientInitializer(EbamDataLogger logger, DataFile dataFile, int recordIndex) {
27      this.logger = logger;
28      this.dataFile = dataFile;
29      this.recordIndex = recordIndex;
30    }
31  
32    @Override
33    public void initChannel(SocketChannel ch) throws IOException {
34      ChannelPipeline pipeline = ch.pipeline();
35  
36      // Decoders
37      pipeline.addLast(new LineBasedFrameDecoder(1024, true, true));
38      pipeline.addLast(new StringDecoder(CharsetUtil.US_ASCII));
39      pipeline.addLast("readTimeoutHandler", new ReadTimeoutHandler(10));
40      pipeline.addLast(new EbamHandler(logger, dataFile, recordIndex));
41  
42      // Encoders
43      pipeline.addLast(new StringEncoder());
44    }
45  }

12  import org.apache.commons.csv.CSVRecord;

14  import java.io.File;
15  import java.text.ParseException;
16  import java.util.Date;
17  import java.util.Iterator;
18  import java.util.List;
19  
20  /**
21   * A class to hold configuration and utility methods for a single data logger.
22   *
23   * @author Tom Parker
24   */
25  public final class EbamDataLogger extends DataLogger {
26  
27    /** column index of date field */
28    public static final int DATE_COLUMN = 0;
29  
30    /** format of date field */
31    public static final String DATE_FORMAT_STRING = "dd-MMM-yyyy hh:mm:ss";
32  
33    /** number of header lines */
34    public static final int HEADER_COUNT = 1;
35  
36    /**
37     * A data model for working with a MetOne E-BAM.
38     *
39     * @param config logger configuration
40     * @throws LoggerException when there are no tables configured
41     */
42    public EbamDataLogger(ConfigFile config) throws LoggerException {
43      super(config, HEADER_COUNT);
44    }
45  
46    /**
47     * Return a filename pattern suitable for passing to SimpleDateFormat.
48     * 
49     * @param dataFile
50     * 
51     * @param table logger table
52     * @return String suitable for SimpleDateFormat
53     */
54    public String getFilePattern(DataFile dataFile) {
55      final StringBuilder sb = new StringBuilder();
56      sb.append("'" + pathRoot + "/" + name + "/'");
57      sb.append(filePathFormat.toPattern());
58      sb.append("'/" + name + "-" + dataFile.toString() + "'");
59      sb.append(fileSuffixFormat.toPattern());
60  
61      final String filename = sb.toString().replace('/', File.separatorChar);
62      return filename;
63    }
64  
65  
66    /**
67     * Find a date in the record.
68     * 
69     * @param record record to search
70     * @return the Date found
71     * @throws ParseException when format cannot be parsed
72     */
73    protected Date parseDate(CSVRecord record) throws ParseException {
74      String dateString = record.get(DATE_COLUMN);
75  
76      return dateFormat.parse(dateString);
77    }
78  }
